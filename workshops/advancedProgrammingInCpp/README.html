           std::vector<int>  list  (10);
            ^     ^     ^      ^     ^
        Namespace |     |      |   Size
                Class   |   Object
                      Type

Defining Namespaces

#include <vector>
#include <string>

namespace numbers {
  // Define functions
  void sort(std::vector<int> &array) {
    // A sorting function
  }

  // Define variables
  std::string programName;

  // Define classes
  struct student { };
}

Accessing from a namespace

  • prefixing numbers:: before their names: 

    numbers::sort(list);

  • Copy everything in the global namespace: 

    using namespace numbers;
sort(list);
